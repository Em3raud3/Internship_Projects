from pathlib import Path
import csv
from flask import Flask, render_template
app = Flask(__name__)



@app.route('/')
def loadAccessFile():
    
    logList = []
    uniqueID = set()
    uniqueDate = set()
    
    fr = open("AccessLog.txt", "rt")
    
    for x in fr:
        tempList = x.split()
        logList.append([tempList[2], ((tempList[3])[1:]), tempList[8], tempList[2] + ((tempList[3])[1:])])
        uniqueID.add(tempList[2] + ((tempList[3])[1:]))
        uniqueDate.add(tempList[8])
    
    fr.close()   
    
    Database = Path("static/js/Database.csv")
    Database.touch(exist_ok=True)
    
    with open("static/js/Database.csv", 'w', newline='') as Database:
        dataWriter = csv.writer(Database)
        
        dataWriter.writerow(["UserID","Date", "Daily Logins", "Failed_Attempts"])
        
        for x in uniqueID:
            length = len(x)
            loginCount = 0
            failCount = 0
            name = x[0:length + -11]
            date = x[length + -11:]
        
            for y in logList:
                if y[3] == x and y[2] == "200":
                    loginCount += 1
                elif y[3] == x and y[2] == "401":
                    failCount += 1
        
            
            dataWriter.writerow([name, date, loginCount, failCount])     
            
            
    Database = Path("static/js/AggegrateDailyLogin.csv")
    Database.touch(exist_ok=True)      
    
    with open("static/js/AggegrateDailyLogin.csv", 'w', newline='') as DailyLogin:
        dataWriter = csv.writer(DailyLogin)
        
        dataWriter.writerow(["Date", "Daily Logins"])
        
        for x in uniqueDate:
            length = len(x)
            loginCount = 0
            failCount = 0
            
            for y in logList:
                if y[1] == x and y[2] == "200":
                    loginCount += 1
                elif y[1] == x and y[2] == "401":
                    failCount += 1
        
            
            dataWriter.writerow([x, loginCount])  
    
    return render_template("index.html")
